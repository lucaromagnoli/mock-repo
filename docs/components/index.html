UI Component Library

A collection of accessible, customizable UI components built with vanilla JavaScript and CSS. Designed for ease of use, accessibility, and consistency across your web applications.

Quick Start

Get started by including the CSS and JavaScript files in your HTML.

HTML 
<!-- Include styles -->
<link rel="stylesheet" href="src/styles/main.css">

<!-- Include component scripts -->
<script src="src/components/Button/Button.js"></script>
<script src="src/components/Input/validators.js"></script>
<script src="src/components/Input/Input.js"></script>
JavaScript 
// Create a button
const button = new Button({
  text: 'Click me',
  variant: 'primary',
  onClick: () => alert('Hello!')
});
document.body.appendChild(button.render());

// Create an input with validation
const input = new Input({
  type: 'email',
  name: 'email',
  label: 'Email Address',
  required: true,
  validators: [validators.email()]
});
document.body.appendChild(input.render());
